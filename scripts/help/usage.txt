BioYoda v0.2.0 - Biomedical data processing pipeline

Usage: bioyoda.sh <command> [options]

COMMANDS:
    start                     Start Qdrant server
    stop                      Stop Qdrant server
    enju <module>             Run data processing as an Enju DAG (preferred)
    run <module>              Run data processing via local Snakemake
    status                    Show pipeline status
    qdrant <subcommand>       Manage Qdrant vector database
    test [--pipeline]         Run test suite
    snapshot --name <name>    Create isolated code snapshot
    validate <module>         Validate module outputs
    clean [module]            Clean intermediate files
    dryrun <module>           Show what would be executed
    dag <module>              Generate workflow DAG visualization
    push <module>             Sync data/code to Google Drive (for Colab GPU)
    pull <module>             Pull processed results from Google Drive
    unlock                    Unlock working directory after crash
    help                      Show this help message
    version                   Show version information

DATA PROCESSING MODULES:
    pubmed                    PubMed literature (30M+ abstracts)
    clinical_trials           Clinical trials data (500K+ trials)
    patents                   Patents (43M+ patents, 30M+ compounds)
    esm2                      Protein embeddings (ESM-2 model)
    diamond                   Protein sequence similarity (DIAMOND BLASTP)
    all                       Run all modules

QDRANT SUBCOMMANDS:
    start                     Start Qdrant server (local or cluster)
    stop                      Stop Qdrant server
    restart [path]            Restart with existing storage (auto-detect or specify path)
    status                    Check server status and collections
    insert <dataset>          Insert data (pubmed, clinical_trials, patents, esm2, all)
    stop-insert <dataset>     Stop a running insertion job

RUN OPTIONS:
    --local                   Run locally (default; only mode)
    --bg, -b                  Run in background with nohup
    --cores N                 Number of cores (default: 1)
    --config FILE             Use custom config file
    --test                    Use test configuration
    --mode <full|update>      Processing mode (default: full)
    --dryrun                  Show what would be executed
    --compounds-only          Process only compounds (patents module, CPU optimized)
    --text-only               Process only text embeddings (patents module)
    --biobtree                Convert patents to JSON for biobtree ingestion

QDRANT START OPTIONS:
    --runtime <hours>         Runtime in hours (default: 48)
    --memory <mb>             Memory in MB (default: 32000)
    --out-dir <path>          Use existing storage directory

PUSH/PULL OPTIONS:
    --code-only, -c           Only sync scripts (fast, for code changes)
    --dry-run, -n             Show what would be synced without syncing
    --bg, -b                  Run in background

EXAMPLES:
    # Quick start/stop servers
    bioyoda.sh start
    bioyoda.sh stop

    # Data processing (Enju DAG — preferred)
    bioyoda.sh enju pubmed
    bioyoda.sh enju all

    # Or plain local Snakemake
    bioyoda.sh run pubmed --local
    bioyoda.sh run pubmed --test --local

    # Patents: separate text vs compounds processing
    bioyoda.sh run patents --local --cores 8 --compounds-only   # CPU (RDKit fingerprints)
    bioyoda.sh run patents --local --cores 4 --text-only        # Or use Colab GPU
    bioyoda.sh run patents --local --biobtree                   # Convert to biobtree JSON

    # Qdrant operations
    bioyoda.sh qdrant start                                  # Start locally (Singularity)
    bioyoda.sh qdrant start --out-dir /scratch/qdrant_data  # Use existing storage
    bioyoda.sh qdrant restart                                # Auto-detect and restart
    bioyoda.sh qdrant insert pubmed                          # Insert data (local)
    bioyoda.sh qdrant status
    bioyoda.sh qdrant stop

    # Testing
    bioyoda.sh test                    # Fast fixture mode
    bioyoda.sh test --pipeline         # Full E2E test

    # Production snapshot
    bioyoda.sh snapshot --name prod_v1.0
    cd snapshots/prod_v1.0/code
    ./bioyoda.sh enju all

    # Google Drive sync (for Colab GPU processing)
    bioyoda.sh push patents                     # Push data + code to Drive
    bioyoda.sh push patents --code-only         # Push only scripts (fast)
    bioyoda.sh push clinical_trials             # Push clinical trials data
    bioyoda.sh push esm2                        # Push ESM2 protein data
    bioyoda.sh push pubmed --dry-run            # Preview what would sync
    bioyoda.sh pull patents                     # Pull processed results
    bioyoda.sh pull clinical_trials             # Pull clinical trials results
    bioyoda.sh pull esm2                        # Pull ESM2 embeddings

    # Maintenance
    bioyoda.sh status
    bioyoda.sh stop pubmed --clean
    bioyoda.sh unlock
